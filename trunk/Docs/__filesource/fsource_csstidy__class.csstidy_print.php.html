    115. 

  115.      * Returns the formatted CSS Code and saves it into $this->output_css and $this->output_css_plain
    116.

    117. 

  117.      * @access private

  120.     function _print($plain = false)
    121. 

  121.     {
    122. 

  122.         if ($this->output_css && $this->output_css_plain) {
    123. 

  123.             return;
    124. 

  124.         }
    125. 

  125.         
    126. 

  126.         $output = '';
    127. 

  127.         if (!$this->parser->get_cfg('preserve_css')) {
    128. 

  128.             $this->_convert_raw_css();
    129. 

  129.         }
    130. 

  130.  
    131. 

  131.         $template =& $this->template;
    132. 

  132.  
    133. 

  133.         if ($plain) {
    134. 

  134.             $template = array_map('strip_tags', $template);
    135. 

  135.         }
    136. 

  136.         
    137. 

  137.         if ($this->parser->get_cfg('timestamp')) {
    138. 

  138.             array_unshift($this->tokens, array(COMMENT, ' CSSTidy ' . $this->parser->version . ': ' . date('r') . ' '));
    139. 

  139.         }
    140. 

  140.         
    141. 

  141.         if (!empty($this->charset)) {
    142. 

  142.             $output .= $template[0].'@charset '.$template[5].$this->charset.$template[6];
    143. 

  143.         }
    144. 

  144.         
    145. 

  145.         if (!empty($this->import)) {
    146. 

  146.             for ($i = 0, $size = count($this->import); $i < $size; $i ++) {
    147. 

  147.                 $output .= $template[0].'@import '.$template[5].$this->import[$i].$template[6];
    148. 

  148.             }
    149. 

  149.         }
    150. 

  150.         
    151. 

  151.         if (!empty($this->namespace)) {
    152. 

  152.             $output .= $template[0].'@namespace '.$template[5].$this->namespace.$template[6];
    153. 

  153.         }
    154. 

  154.         
    155. 

  155.         $output .= $template[13];
    156. 

  156.         $in_at_out = '';
    157. 

  157.         $out =& $output;
    158. 

  158.         
    159. 

  159.         foreach ($this->tokens as $key => $token)
    160. 

  160.         {
    161. 

  161.             switch ($token[0])
    162. 

  162.             {
    163. 

  163.                 case AT_START:
    164. 

  164.                     $out .= $template[0].$this->_htmlsp($token[1], $plain).$template[1];
    165. 

  165.                     $out =& $in_at_out;
    166. 

  166.                     break;
    167. 

  167.                 
    168. 

  168.                 case SEL_START:
    169. 

  169.                     if($this->parser->get_cfg('lowercase_s')) $token[1] = strtolower($token[1]);
    170. 

  170.                     $out .= ($token[1]{0} !== '@') ? $template[2].$this->_htmlsp($token[1], $plain) : $template[0].$this->_htmlsp($token[1], $plain);
    171. 

  171.                     $out .= $template[3];
    172. 

  172.                     break;
    173. 

  173.                     
    174. 

  174.                 case PROPERTY:
    175. 

  175.                     if($this->parser->get_cfg('case_properties') == 2) $token[1] = strtoupper($token[1]);
    176. 

  176.                     if($this->parser->get_cfg('case_properties') == 1) $token[1] = strtolower($token[1]);
    177. 

  177.                     $out .= $template[4] . $this->_htmlsp($token[1], $plain) . ':' . $template[5];
    178. 

  178.                     break;
    179. 

  179.                 
    180. 

  180.                 case VALUE:
    181. 

  181.                     $out .= $this->_htmlsp($token[1], $plain);
    182. 

  182.                     if($this->_seeknocomment($key, 1) == SEL_END && $this->parser->get_cfg('remove_last_;')) {
    183. 

  183.                         $out .= str_replace(';', '', $template[6]);
    184. 

  184.                     } else {
    185. 

  185.                         $out .= $template[6];
    186. 

  186.                     }
    187. 

  187.                     break;
    188. 

  188.                 
    189. 

  189.                 case SEL_END:
    190. 

  190.                     $out .= $template[7];
    191. 

  191.                     if($this->_seeknocomment($key, 1) != AT_END) $out .= $template[8];
    192. 

  192.                     break;
    193. 

  193.                 
    194. 

  194.                 case AT_END:
    195. 

  195.                     $out =& $output;
    196. 

  196.                     $out .= $template[10] . str_replace("\n", "\n" . $template[10], $in_at_out);
    197. 

  197.                     $in_at_out = '';
    198. 

  198.                     $out .= $template[9];
    199. 

  199.                     break;
    200. 

  200.  
    201. 

  201.                 case COMMENT:
    202. 

  202.                     $out .= $template[11] . '/*' . $this->_htmlsp($token[1], $plain) . '*/' . $template[12];
    203. 

  203.                     break;
    204. 

  204.             }
    205. 

  205.         }
    206. 

  206.  
    207. 

  207.         $output = trim($output);
    208. 

  208.         
    209. 

  209.         if (!$plain) {
    210. 

  210.             $this->output_css = $output;
    211. 

  211.             $this->_print(true);
    212. 

  212.         } else {
    213. 

  213.             $this->output_css_plain = $output;
    214. 

  214.         }
    215. 

  215.     }

  217.     /**
    218. 

  218.      * Gets the next token type which is $move away from $key, excluding comments
    219. 

  219.      * @param integer $key current position
    220. 

  220.      * @param integer $move move this far
    221. 

  221.      * @return mixed a token type

  222.      * @access private

  225.     function _seeknocomment($key, $move) {
    226. 

  226.         $go = ($move > 0) ? 1 : -1;
    227. 

  227.         for ($i = $key + 1; abs($key-$i)-1 < abs($move); $i += $go) {
    228. 

  228.             if (!isset($this->tokens[$i])) {
    229. 

  229.                 return;
    230. 

  230.             }
    231. 

  231.             if ($this->tokens[$i][0] == COMMENT) {
    232. 

  232.                 $move += 1;
    233. 

  233.                 continue;
    234. 

  234.             }
    235. 

  235.             return $this->tokens[$i][0];
    236. 

  236.         }
    237.

  244.     function _convert_raw_css()
    245. 

  245.     {
    246. 

  246.         $this->tokens = array();

  247.         ksort($this->css);
    248. 

  248.         
    249. 

  249.         foreach ($this->css as $medium => $val)
    250. 

  250.         {
    251. 

  251.             if ($this->parser->get_cfg('sort_selectors')) ksort($val);
    252. 

  252.             if ($medium != DEFAULT_AT) {
    253. 

  253.                 $this->parser->_add_token(AT_START, $medium, true);
    254. 

  254.             }
    255. 

  255.             
    256. 

  256.             foreach ($val as $selector => $vali)
    257. 

  257.             {
    258. 

  258.                 if ($this->parser->get_cfg('sort_properties')) ksort($vali);
    259. 

  259.                 $this->parser->_add_token(SEL_START, $selector, true);
    260. 

  260.                 
    261. 

  261.                 foreach ($vali as $property => $valj)
    262. 

  262.                 {
    263. 

  263.                     $this->parser->_add_token(PROPERTY, $property, true);
    264. 

  264.                     $this->parser->_add_token(VALUE, $valj, true);
    265. 

  265.                 }
    266. 

  266.                 
    267. 

  267.                 $this->parser->_add_token(SEL_END, $selector, true);
    268. 

  268.             }
    269. 

  269.             
    270. 

  270.             if ($medium != DEFAULT_AT) {
    271. 

  271.                 $this->parser->_add_token(AT_END, $medium, true);
    272. 

  272.             }
    273. 

  273.         }
    274. 

  274.     }
